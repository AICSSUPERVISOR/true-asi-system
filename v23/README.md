### Agent Swarm
```typescript
import { agentSwarm } from './frontend/agent_swarm';

// Create agent
const agent = agentSwarm.createAgent('researcher');

// Assign task
await agentSwarm.assignTask(agent.id, { type: 'research', query: '...' });

// Evolve swarm
agentSwarm.evolve();
```
